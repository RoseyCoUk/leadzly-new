    JS EDIT — Add new IIFE as a SEPARATE <script> block in index.html, immediately before the closing </body> tag and AFTER the existing closing </script> tag:

    ```html
    <script>
    // FAQ Accordion
    (function() {
      var items = document.querySelectorAll('.faq__item');
      items.forEach(function(item) {
        var btn = item.querySelector('.faq__question');
        if (!btn) return;
        btn.addEventListener('click', function() {
          var isActive = item.classList.contains('faq__item--active');
          // collapse all
          items.forEach(function(i) {
            i.classList.remove('faq__item--active');
            var b = i.querySelector('.faq__question');
            if (b) b.setAttribute('aria-expanded', 'false');
          });
          // open clicked (unless it was already open)
          if (!isActive) {
            item.classList.add('faq__item--active');
            btn.setAttribute('aria-expanded', 'true');
          }
        });
      });
    })();
    </script>
    ```

    CRITICAL: Do NOT merge this into the existing script block. Add it as a new standalone <script> block after the existing </script> closing tag.
    Do NOT use any external JS libraries — vanilla only.
    Do NOT add `type="module"` — plain `<script>` tag only (IIFE handles scope).
